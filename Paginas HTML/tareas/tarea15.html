Tarea 15

Problema PS3.5

Construye un programa que calcule e imprima la productoria de los N primeros números naturales.
N sumatoria PI i desde i = 1

JAVA

import javax.swing.JOptionPane;
public class tarea15{
	public static void main (String[] args){
		int i, n;
		double a;
		String entrada;
		String serie = "";

		entrada = JOptionPane.showInputDialog("Ingrese el numero de terminos: ");
		n = Integer.parseInt(entrada);

		for(i=1; i< n; i++){
			a = 3.1416*i;
			serie = serie + a + ", ";
		}
		a = 3.1416*i;
		serie = serie + a + ".";
		JOptionPane.showMessageDialog(null,serie);
	} //Fin de función main
} //Fin de clase tarea15
