Problema PS5.3

Escribe un programa que almacene en un arreglo unidimensional los primeros 100 números de Fibonacci e imprima el arreglo correspondiente.

JAVA

import javax.swing.JOptionPane;
public class tarea19{
	public static void main (String[] args){
		int fibo=1, i,a=1,b=0;
		int[] vector = new int[100];
		String serie = " . ";

		for(i=1; i< 100; i++){
			fibo = a + b;
			a = b;
			b = fibo;

			vector[i] = fibo;
		}

		for(i=1; i< 100; i++){
			serie = serie + vector[i] + " . ";
		}
		JOptionPane.showMessageDialog(null,"["+serie+"]");
	} //Fin de función main
} //Fin de clase tarea19
