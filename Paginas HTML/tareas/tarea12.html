Tarea 12

Construir un programa que calcule e imprima la siguiente serie para los N primeros términos:
-1, 2, 2, -3, 4, 4, 4, 4, -5, 6, 6, 6, 6, -7, ...

JAVA

import javax.swing.JOptionPane;
public class tarea12{
	public static void main (String[] args){
		String entrada;
		String serie = "";
		int n,i,y=1,x=0,z=1,a=2;

		entrada = JOptionPane.showInputDialog("Ingrese el numero de terminos: ");
		n = Integer.parseInt(entrada);

		if(n>0){
			serie = serie + "-1";

			for(i=1; i< n; i++){
				serie = serie + ", ";
				if(z==0){
					serie = serie + (a*-1);
					z = 1;
					x = 0;
					a++;
				}else if(z==1){
					serie = serie + a;
					if(x< y){
						x++;
					}else{
						z = 0;
						a++;
						y++;
					}
				}
			}
		}
		JOptionPane.showMessageDialog(null,serie);
	} //Fin de función main
} //Fin de clase tarea12
